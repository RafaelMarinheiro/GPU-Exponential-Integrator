00015 // This simply initializes it to random values.
00016 // Note that A is stored column-wise, not row-wise.
00017 //
00018 // m   - number of rows,    m >= 0.
00019 // n   - number of columns, n >= 0.
00020 // A   - m-by-n array of size lda*n.
00021 // lda - leading dimension of A, lda >= m.
00022 //
00023 // When lda > m, rows (m, ..., lda-1) below the bottom of the matrix are ignored.
00024 // This is helpful for working with sub-matrices, and for aligning the top
00025 // of columns to memory boundaries (or avoiding such alignment).
00026 // Significantly better memory performance is achieved by having the outer loop
00027 // over columns (j), and the inner loop over rows (i), than the reverse.
00028 void zfill_matrix(
00029     magma_int_t m, magma_int_t n, magmaDoubleComplex *A, magma_int_t lda )
00030 {
00031     #define A(i_, j_) A[ (i_) + (j_)*lda ]
00032     
00033     magma_int_t i, j;
00034     for( j=0; j < n; ++j ) {
00035         for( i=0; i < m; ++i ) {
00036             A(i,j) = MAGMA_Z_MAKE( rand() / ((double) RAND_MAX),    // real part
00037                                    rand() / ((double) RAND_MAX) );  // imag part
00038         }
00039     }
00040 }
